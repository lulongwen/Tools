      function formatDate() {
        let DATE = new Date();
        let years = DATE.getFullYear();
        let month = ('0' + (DATE.getMonth() + 1)).slice(-2);
        let day = ('0' + DATE.getDate()).slice(-2);
        return `${years}-${month}-${day}`; 
      }
